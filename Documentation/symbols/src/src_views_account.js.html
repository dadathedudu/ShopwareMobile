208 	/**
209 	 * onBackBtn
210 	 *
211 	 * Handles the back button
212 	 *
213 	 * @param btn
214 	 * @param event
215 	 */
216 	onBackBtn: function(btn, event) {
217         var curr      = this.getActiveItem(),
218             currIdx   = this.items.indexOf(curr);
219 
220         if (currIdx != 0) {
221             var prevDepth     = currIdx - 1,
222                 prev          = this.items.getAt(prevDepth);
223 
224             this.setActiveItem(prev, {
225                 type: 'slide',
226                 reverse: true,
227                 scope: this
228             });
229             this.syncToolbar(prev);
230 
231 	        if(curr) { curr.destroy(); }
232         }
233     },
234 
235 	/**
236 	 * syncToolbar/tapped
237 	 *
238 	 * @param card - active card
239 	 */
240 	syncToolbar: function(card) {
241 		var active        = card || this.getActiveItem(),
242 			depth         = this.items.indexOf(active),
243 			backBtn       = this.backBtn,
244 			backToggleMth = (depth !== 0) ? 'show' : 'hide',
245 			title;
246 
247 		if(active.title) {
248 			title = active.title;
249 		} else if(active.ownerCt.title) {
250 			title = active.ownerCt.title;
251 		}
252 
253 		if(title.length) {
254 			this.toolbar.setTitle(title);
255 		}
256 
257 	/**
258 	 * Refreshes the toolbar after the back btn
259 	 * was clicked
260 	 */
261 		if(depth === 0) {
262 			this.backBtn.hide();
263 			this.toolbar.doLayout();
264 		}
265 	}

286 	items: [{
287 		xtype: 'fieldset',
288 		title: 'Login',
289 		instructions: 'Bitte geben Sie Ihre Kundendaten ein',
290 		defaults: { labelWidth: '38%' },
291 		items: [
292 			{
293 				xtype: 'hiddenfield',
294 				name: 'accountmode',
295 				value: 2
296 			},
297 			{
298 				xtype: 'emailfield',
299 				name: 'email',
300 				label: 'E-Mail',
301 				required: true,
302 				placeHolder: 'me@shopware.de'
303 			},
304 			{
305 				xtype: 'passwordfield',
306 				name: 'password',
307 				label: 'Passwort',
308 				required: true
309 			}
310 		]
311 	}, {
312 		id: 'loginBtn',
313 		xtype: 'button',
314 		text: 'Login absenden',
315 		style: 'margin: 0.5em',
316 		ui: 'confirm',
317 		scope: this,
318 		handler: this.onLoginBtn
319 	}],
320 	listeners: {
321 		scope: this,
322 		deactivate: function(me) {
323 			me.destroy();
324 		},
325 		submit: function(form, response) {
326 			if(response.success && response.msg) {
327 				Ext.Msg.alert('Login erfolgreich', response.msg, function() {
328 					var active = Ext.getCmp('viewport').getActiveItem(),
329 						view = 0;
330 
331 					if(active.id == 'cart') {
332 						App.stores.UserData.load({
333 							scope: this,
334 							callback: function(){
335 								view = new App.views.Checkout.index;
336 								view.update('');
337 								active.setActiveItem(view, {
338 									type: 'slide',
339 									reverse: true,
340 									scope: this
341 								});
342 							}
343 						});
344 					} else {
345 
346 
347 						/* Toolbar set up */
348 						active.backBtn.hide();
349 						active.logoutBtn.show();
350 						active.toolbar.setTitle(active.backBtn.text);
351 						active.toolbar.doLayout();
352 
353 						/* Change active view */
354 						active.newPnl.hide();
355 						active.existingPnl.hide();
356 
357 						App.stores.UserData.load({
358 							scope: this,
359 							callback: function() {
360 								active.createCustomerCenter(active.mainPnl);
361 								active.mainPnl.doComponentLayout();
362 								active.mainPnl.doLayout();
363 
364 								active.setActiveItem(view, {
365 									type: 'slide',
366 									reverse: true,
367 									scope: this
368 								});
369 							}
370 						})
371 					}
372 				});
373 			}
374 		},
375 		exception: function(form, response) {
376 			if(!response.success && response.msg) {
377 				Ext.Msg.alert('Login fehlgeschlagen', response.msg);
378 			}
379 		}
380 	},
381 
382 	initComponent: function() {
383 		App.views.Account.login.superclass.initComponent.call(this);
384 		Ext.getCmp('loginBtn').on('tap', this.onLoginBtn, this);
385 	},
386 
387 	onLoginBtn: function() {
388 		this.submit();
389 	}
