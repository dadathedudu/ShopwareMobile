  7 Ext.ns('App.views.Viewport', 'App.views.Shop', 'App.views.Search', 'App.views.Cart', 'App.views.Account', 'App.views.Info', 'App.views.Checkout');
  8 
  9 /**
 10  * Main checkout view
 11  *
 12  * Contains all necessary checkout components
 13  *
 14  * @access public
 15  * @class
 16  * @extends Ext.Panel
 17  */
 18 App.views.Checkout.index = Ext.extend(Ext.Panel,
 19 /** @lends App.views.Checkout.index# */
 20 {
 21 	id: 'orderConfirmation',
 22 	title: 'Bestellbestätigung',
 23 	scroll: 'vertical',
 24 	layout: 'card',
 25 	listeners: {
 26 		scope: this,
 27 		deactivate: function(me) {
 28 			console.log('destroy orderConfirmation');
 29 			me.destroy();
 30 		}
 31 	},
 32 	
 33 	initComponent: function() {
 34 		var me = this,
 35 			userData = App.stores.UserData.proxy.reader.rawData.sUserData,
 36 			billing  = userData.billingaddress,
 37 			shipping = userData.shippingaddress,
 38 			dispatch = userData.activeDispatch;
 39 
 40 		/** Back btn */
 41 		this.backBtn = new Ext.Button({
 42 			text: 'Warenkorb',
 43 			ui: 'back',
 44 			scope: this,
 45 			handler: this.onBackBtn
 46 		});
 47 
 48 		/** Toolbar */
 49 		this.toolbar = new Ext.Toolbar({
 50 			title: App.Helpers.truncate(this.title, 14),
 51 			items: [this.backBtn]
 52 		});
 53 
 54 		/** Cart listing */
 55 		this.cartLabel = new Ext.Container({
 56 			id: 'cartListLabel',
 57 			html: '<div class="x-form-fieldset-title">Ihr Warenkorb</div>'
 58 		});
 59 		this.cartView = new App.views.Cart.list;
 60 		this.cartView.tpl = App.views.Checkout.cartTpl;
 61 
 62 
 63 		/** Payment methods fieldset */
 64 		this.paymentField = new Ext.form.FieldSet({
 65 			title: 'Ausgewählte Zahlungsart',
 66 			items: [{
 67 				xtype: 'textfield',
 68 				value: userData.additional.payment.description,
 69 				name: 'paymentMeans',
 70 				disabled: true
 71 			}]
 72 		});
 73 
 74 		/** shipping type fieldset */
 75 		this.shippingTypeField = new Ext.form.FieldSet({
 76 			id: 'shippingTypeField',
 77 			title: 'Ausgewählte Versandart',
 78 			items: [{
 79 				xtype: 'textfield',
 80 				value: dispatch.name,
 81 				checked: true,
 82 				name: 'defaultDispatch',
 83 				disabled: true
 84 			}]
 85 		});
 86 
 87 		/** General checkout form pnl */
 88 		this.formPnl = new Ext.form.FormPanel({
 89 			height: '100%',
 90 			scroll: false,
 91 			id: 'checkoutForm',
 92 			items: [this.paymentField, this.shippingTypeField]
 93 		});
 94 
 95 		/** Billing address */
 96 		this.billing = new Ext.Container({
 97 			id: 'billingAddress',
 98 			html: '<div class="label x-form-fieldset-title">Rechnungsadresse</div>' +
 99 				  '<div class="inner">' +
100 				    '<p>' + billing.firstname + ' ' + billing.lastname + '</p>' +
101 					'<p>' + billing.street + ' ' + billing.streetnumber + '</p>' +
102 					'<p>' + billing.zipcode + ' ' + billing.city + '</p>' +
103 				  '</div>'
104 		});
105 
106 		/** Shipping address */
107 		this.shipping = new Ext.Container({
108 			id: 'shippingAddress',
109 			html: '<div class="label x-form-fieldset-title">Lieferadresse</div>' +
110 				  '<div class="inner">' +
111 				    '<p>' + shipping.firstname + ' ' + shipping.lastname + '</p>' +
112 					'<p>' + shipping.street + ' ' + shipping.streetnumber + '</p>' +
113 					'<p>' + shipping.zipcode + ' ' + shipping.city + '</p>' +
114 				  '</div>'
115 		});
116 
117 		/** Handles all neccessary order informations */
118 		this.orderPnl = new Ext.form.FormPanel({
119 			id: 'orderPnl',
120 			url: App.RequestURL.saveOrder,
121 			listeners: {
122 				beforesubmit: function() {
123 					this.ownerCt.setLoading(true);
124 				},
125 				submit: function(form, response) {
126 					this.ownerCt.setLoading(false);
127 					if(response.success && response.msg) {
128 						Ext.Msg.alert('Bestellung erfolgreich', response.msg, function() {
129 							App.stores.Cart.removeAll();
130 
131 							/* Could be cleaner... */
132 							me.setActiveItem(0);
133 							me.toolbar.show();
134 							me.doComponentLayout();
135 							Ext.getCmp('viewport').setActiveItem(0, {
136 								type: 'slide',
137 								reverse: true,
138 								scope: this
139 							})
140 						});
141 					}
142 				},
143 
144 				exception: function(form, response) {
145 					if(!response.success && response.msg) {
146 						Ext.Msg.alert('Bestellung fehlgeschlagen', response.msg);
147 						form.ownerCt.setLoading(false);
148 					}
149 				}
150 			}
151 		});
152 
153 		/** Order comment */
154 		if(~~useComment) {
155 			this.commentField = new Ext.form.FieldSet({
156 				id: 'commentFieldset',
157 				title: 'Ihr Kommentar',
158 				instructions: 'Bitte geben Sie hier Ihr Kommentar zu Ihrer Bestellung ein.',
159 				items: [{
160 					xtype: 'textareafield',
161 					name: 'sComment',
162 					label: 'Kommentar',
163 					value: ''
164 				}]
165 			});
166 			this.orderPnl.add(this.commentField);
167 		}
168 
169 		/** Voucher field */
170 		if(~~useVoucher) {
171 			this.voucherField = new Ext.form.FieldSet({
172 				title: 'Gutschein hinzufügen',
173 				instructions: 'Möchten Sie einen Gutschein zu Ihrer Bestellung hinzufügen?',
174 				items: [{
175 					xtype: 'textfield',
176 					label: 'Gutschein',
177 					name: 'sVoucher'
178 				}]
179 			});
180 			this.orderPnl.add(this.voucherField);
181 		}
182 
183 		/** Subscribe to newsletter */
184 		if(~~useNewsletter) {
185 			this.newsletterField = new Ext.form.FieldSet({
186 				id: 'newsletterFieldset',
187 				title: 'Newsletter abonnieren',
188 				instructions: 'M�chten Sie den kostenlosen ' + shopName + ' Newsletter erhalten? Sie k�nnen sich jederzeit wieder abmelden! ',
189 				items: [{
190 					xtype: 'checkboxfield',
191 					name: 'sNewsletter',
192 					label: 'Newsletter',
193 					checked: false,
194 					value: 'active'
195 				}]
196 			});
197 			this.orderPnl.add(this.newsletterField);
198 		}
199 
200 		/** AGB checkbox */
201 		this.agbField = new Ext.form.FieldSet({
202 			id: 'agbFieldset',
203 			title: 'AGBs akzeptieren',
204 			instructions: 'Bitte bestätigen Sie die geltenden Allgemeinen Geschäftsbedingungen.',
205 			items: [{
206 				xtype: 'checkboxfield',
207 				checked: false,
208 				label: 'AGBs',
209 				name: 'sAGB',
210 				required: true,
211 				value: 'active'
212 			}]
213 		});
214 
215 		/** Submit order */
216 		this.submitOrderBtn = new Ext.Button({
217 			id: 'submitOrderBtn',
218 			ui: 'confirm',
219 			text: 'Bestellung absenden',
220 			scope: this,
221 			handler: this.onSubmitOrderBtn
222 		});
223 
224 		this.orderPnl.add(this.agbField);
225 		this.orderPnl.add(this.submitOrderBtn);
226 
227 		/** Main Pnl */
228 		this.pnl = new Ext.Panel({
229 			scroll: 'vertical',
230 			items: [
231 				this.cartLabel,
232 				this.cartView,
233 				this.formPnl,
234 				this.billing,
235 				this.shipping,
236 				this.orderPnl,
237 				this.submitOrderBtn
238 			]
239 		});
240 
241 		/** Apply all components to main pnl */
242 		Ext.apply(this, {
243 			dockedItems: [this.toolbar],
244 			items: [this.pnl]
245 		});
246 		App.views.Checkout.index.superclass.initComponent.call(this);
247 	},
248 
249 	/** onBackBtn - event handler */
250 	onBackBtn: function() {
251 		var cart = Ext.getCmp('cart');
252 		cart.setActiveItem(0, {
253 			type: 'slide',
254 			reverse: true,
255 			scope: this
256 		});
257 		cart.toolbar.show();
258 		cart.doComponentLayout();
259 	},
260 
261 	/** onSubmitOrderBtn - event handler */
262 	onSubmitOrderBtn: function() {
263 		var pnl    = this.orderPnl,
264 			values = pnl.getValues();
265 
266 		if(Ext.isEmpty(values.sAGB)) {
267 			Ext.Msg.alert('Fehler', 'Bitte bestätigen Sie die AGBs um Ihre Bestellung durchzuühren.');
268 			return false;
269 		}
270 
271 		pnl.submit();
272 	}
273 });
