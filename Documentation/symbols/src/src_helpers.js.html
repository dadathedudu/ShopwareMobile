 44     /**
 45      * Sends an general request to the server
 46      *
 47      * @private
 48      * @param url
 49      * @param method
 50      * @param params
 51      * @param callback
 52      * @param scope
 53      */
 54 	_request: function(url, method, params, callback, scope) {
 55 		scope = scope || this;
 56 		Ext.Ajax.request({
 57 			url: url,
 58 			method: method,
 59 			disableCaching: false,
 60 			params: params,
 61 			success: function(response, options) {
 62 				try {
 63 					this.data = Ext.util.JSON.decode(response.responseText)
 64 				} catch(err) {
 65 					Ext.Msg.alert('Fehler', 'Es ist ein Fehler bei einen AJAX-Request aufgetreten, bitte versuchen Sie es späer erneut.');
 66 				}
 67 				
 68 				if(Ext.isFunction(callback)) {
 69 					callback.call(scope, this.data);
 70 				}
 71 			}
 72 		});
 73 	},

 81 	isUserLoggedIn: function() {
 82 		this.getRequest(App.RequestURL.userLoggedIn, '', function(data) {

 84 				this.getUserData();
 85 				isUserLoggedIn = '1';

 87 				isUserLoggedIn = '';
 88 			}
 89 			return data;

165         /**
166          * Initialize the liveshopping
167          *
168          * @param data - basic liveshopping data
169          */
170 		init: function(data) {
171 			var now, diff, target, me = this;
172 			target = new Date();
173 			target.setTime(data.valid_to_ts * 1000);
174 			this._price = null;
175 			
176 			var interval = window.setInterval(function() {
177 				now = App.Helpers.server.dateObj;
178 				diff = App.Helpers.timestampDiff(target.getTime(), now.getTime());
179 
180 				// check difference
181 				if(diff !== false) {
182 
183 					// refresh price
184 					if(data.typeID == 2 || data.typeID == 3) {
185 						App.Helpers.liveshopping.refreshPrices(diff, data);
186 					}
187 
188 					// refresh dates
189 					App.Helpers.liveshopping.refreshDates(diff);
190 				}
191 			}, 1000);
192 
193 			return interval;
194 		},

196         /**
197          * Refreshes the dates and times in the liveshopping display
198          *
199          * @param diff
200          */
201 		refreshDates: function(diff) {
202 			var tmpD = diff.d.toString(), tmpH = diff.h.toString(), tmpM = diff.m.toString(), tmpS = diff.s.toString();
203 
204 			// days
205 			if(tmpD.length == 1) { tmpD = '0' + tmpD }
206 			$$('days_time').innerHTML = tmpD;
207 
208 			// hours
209 			if(tmpH.length == 1) { tmpH = '0' + tmpH }
210 			$$('hours_time').innerHTML = tmpH;
211 
212 			// minutes
213 			if(tmpM.length == 1) { tmpM = '0' + tmpM }
214 			$$('minutes_time').innerHTML = tmpM;
215 
216 			// seconds
217 			if(tmpS.length == 1) { tmpS = '0' + tmpS }
218 			$$('seconds_time').innerHTML = tmpS;
219 
220 			return true;
221 		},

223         /**
224          * Refreshes the prices in the liveshopping display
225          *
226          * @param diff
227          * @param data
228          */
229 		refreshPrices: function(diff, data) {
230 			if(diff.s === 0) {
231 
232 				var newPrice = parseFloat($$('priceHidden').value);
233 				var priceEl = $$('price');
234 
235 				if(data.typeID == 2) {
236 					newPrice = newPrice - data.minPrice;
237 				} else if(data.typeID == 3) {
238 					newPrice = newPrice + data.minPrice;
239 				}
240 
241 				priceEl.innerHTML = App.Helpers.number_format(newPrice, 2, ',', '.') + ' €*';
242 				$$('priceHidden').value = newPrice;
243 			}
244 		}
245 	},

247 	/**
248 	 * truncate
249 	 *
250 	 * Truncates a string and returns the string
251 	 *
252 	 * @param str - string to truncate
253 	 * @param length - length of visible string part
254 	 * @param trailing - trailing symbol
255 	 */
256 	truncate: function(str, length, trailing) {
257 		if(!str || !length) { return false; }
258 
259 		if(!trailing) {
260 			trailing = '...';
261 		}
262 
263 		if(str.length > length) {
264 
265 			length = length - trailing.length;
266 			str = str.substring(0, length) + trailing;
267 		}
268 
269 		return str;
270 	},

272     /**
273      * Formats a number to the european number standard
274      *
275      * @param number
276      * @param decimals
277      * @param dec_point
278      * @param thousands_sep
279      */
280 	number_format: function (number, decimals, dec_point, thousands_sep) {
281         var n = number,
282             prec = decimals;
283         var toFixedFix = function (n, prec) {
284             var k = Math.pow(10, prec);
285             return (Math.round(n * k) / k).toString();
286         };
287         n = !isFinite(+n) ? 0 : +n;prec = !isFinite(+prec) ? 0 : Math.abs(prec);
288         var sep = (typeof thousands_sep === 'undefined') ? ',' : thousands_sep;
289         var dec = (typeof dec_point === 'undefined') ? '.' : dec_point;
290         var s = (prec > 0) ? toFixedFix(n, prec) : toFixedFix(Math.round(n), prec);
291         var abs = toFixedFix(Math.abs(n), prec);
292         var _, i;
293         if (abs >= 1000) {
294             _ = abs.split(/\D/);
295             i = _[0].length % 3 || 3;
296             _[0] = s.slice(0, i + (n < 0)) + _[0].slice(i).replace(/(\d{3})/g, sep + '$1');
297             s = _.join(dec);
298         } else {
299             s = s.replace('.', dec);
300         }
301         var decPos = s.indexOf(dec);
302         if (prec >= 1 && decPos !== -1 && (s.length - decPos - 1) < prec) {
303             s += new Array(prec - (s.length - decPos - 1)).join(0) + '0';
304         } else if (prec >= 1 && decPos === -1) {
305             s += dec + new Array(prec).join(0) + '0';
306         }
307         return s;
308     },

310     /**
311      * Compares two given timestamps and returns their difference
312      *
313      * @param d1
314      * @param d2
315      */
316 	timestampDiff: function (d1, d2) {
317         if (d1 < d2) {
318             return false;
319         }
320         var d = Math.floor((d1 - d2) / (24 * 60 * 60 * 1000));
321         var h = Math.floor(((d1 - d2) - (d * 24 * 60 * 60 * 1000)) / (60 * 60 * 1000));
322         var m = Math.floor(((d1 - d2) - (d * 24 * 60 * 60 * 1000) - (h * 60 * 60 * 1000)) / (60 * 1000));
323         var s = Math.floor(((d1 - d2) - (d * 24 * 60 * 60 * 1000) - (h * 60 * 60 * 1000) - (m * 60 * 1000)) / 1000);
324         return {
325             'd': d,
326             'h': h,
327             'm': m,
328             's': s
329         };
330     }
