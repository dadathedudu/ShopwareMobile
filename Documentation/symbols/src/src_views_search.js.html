  7 Ext.ns('App.views.Viewport', 'App.views.Shop', 'App.views.Search', 'App.views.Cart', 'App.views.Account', 'App.views.Info', 'App.views.Checkout');

  9 /**
 10  * Main search view
 11  *
 12  * Contains all necessary sub components
 13  *
 14  * @access public
 15  * @class
 16  * @extends Ext.Panel
 17  */
 18 App.views.Search.index = Ext.extend(Ext.Panel,
 19 /** @lends App.views.Search.index# */
 20 {
 21 	id: 'search',
 22 	title: 'Suche',
 23 	iconCls: 'search',
 24 	layout: 'card',
 25 	listeners: {
 26 		beforeactivate: function(me) {
 27 			if(me.toolbar.isHidden()) {
 28 				me.toolbar.show();
 29 				me.show();
 30 			}
 31 		}
 32 	},
 33 	initComponent: function() {
 34 
 35 		this.toolbar = new Ext.Toolbar({
 36 			ui: 'dark',
 37 			dock: 'top',
 38 			items: [
 39 				{
 40 					id: 'searchfield',
 41 					xtype: 'searchfield',
 42 					name: 'search_query',
 43 					autoFocus: true,
 44 					hasFocus: true,
 45 					width: '78%',
 46 					placeHolder: 'Ihr Suchbegriff',
 47 					listeners: {
 48 						scope: this,
 49 						keyup: this.onKeyUp
 50 					}
 51 				},
 52 				{ xtype: 'spacer' },
 53 				{
 54 					id: 'searchBtn',
 55 					iconMask: true,
 56 					iconCls: 'search',
 57 					xtype: 'button',
 58 					ui: 'action small',
 59 					scope: this,
 60 					handler: this.onSearch
 61 				}
 62 			]
 63 		});
 64 
 65 		this.emptyHtml = '<div class="emptySearch">'+
 66 			'Bitte geben Sie einen Suchbefehl ein, um die Artikelsuche zu starten.'+
 67 			'</div>';
 68 
 69 		Ext.apply(this, {
 70 			dockedItems: [this.toolbar],
 71 			html: this.emptyHtml
 72 		});
 73 		App.views.Search.index.superclass.initComponent.call(this);
 74 	},
 75 
 76     /**
 77      * Event handler
 78      *
 79      * @param cmp
 80      * @param event
 81      */
 82 	onKeyUp: function(cmp, event) {
 83 		var val = cmp.getValue();
 84 		if (event.browserEvent.keyCode == 13) {
 85 			cmp.blur();
 86 			this.onSearch(val);
 87 		}
 88 	},
 89 
 90     /**
 91      * Event handler
 92      *
 93      * @param val
 94      */
 95 	onSearch: function(val) {
 96 		var list   = Ext.getCmp('searchList'),
 97 			search = Ext.getCmp('search');
 98 		
 99 		if(!list) {
100 			list = new App.views.Search.list();
101 			search.add(list);
102 			search.doLayout();
103 		}
104 		Ext.getCmp('search').add(list);
105 		
106 		if(val.length >= 3) {
107 			Ext.getCmp('search').doLayout();
108 			App.stores.Search.load({
109 				params: {
110 					sSearch: val
111 				}
112 			});
113 
114 			App.stores.Search.proxy.extraParams = {
115 				sSearch: val
116 			};
117 		} else {
118 			list.destroy();
119 			this.update(this.emptyHtml);
120 			this.doLayout();
121 		}
122 	}
123 });
124 
125 /**
126  * Search result list
127  *
128  * @access public
129  * @class
130  * @extends Ext.List
131  */
132 App.views.Search.list = Ext.extend(Ext.List,
133 /** @lends App.views.Search.list */
134 {
135 	id: 'searchList',
136 	store: App.stores.Search,
137 	itemTpl: App.views.Search.itemTpl,
138 	itemtap: this.onItemTap,
139 	scope: this,
140 	initComponent: function() {
141 
142 		this.store.on({
143 			scope: this,
144 			datachanged: this.onDataChanged
145 		});
146 		App.views.Search.list.superclass.initComponent.call(this);
147 	},
148 
149     /**
150      * Event handler
151      * 
152      * @param store
153      */
154 	onDataChanged: function(store) {
155 		if(store.data.items.length >= 1) {
156 			this.refresh();
157 		} else {
158 			this.ownerCt.update('<div class="emptySearch">Es wurden keine Ergebnisse zu Ihren Suchbegriff gefunden.</div>');
159 			this.destroy();
160 		}
161 	},
162 
163     /**
164      * Event handler
165      *
166      * @param view
167      * @param idx
168      * @param item
169      * @param event
170      */
171 	onItemTap: function(view, idx, item, event) {
172 		var tmpRec = this.store.getAt(idx);
173 		Ext.dispatch({
174 			controller: 'detail',
175 			action: 'show',
176 			articleID: tmpRec.data.articleID,
177 			parent: this.ownerCt
178 		})
179 
180 		this.ownerCt.toolbar.hide();
181 	}
182 });
