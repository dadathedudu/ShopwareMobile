675  */
676 App.views.Shop.pictures = Ext.extend(Ext.Carousel,
677 /** @lends App.views.Shop.pictures# */
678 {
679 	id: 'pictures',
680 	direction: 'horizontal',
681 	listeners: {
682 		scope: this,
683 		beforeactivate: function(me) {
684 			me.setLoading(true);
685 		},
686 		activate: function(me) {
687 			me.setLoading(false);
688 		},
689 		deactivate: function(me) {
690 			me.destroy();
691 		}
692 	},
693 	initComponent: function() {
694 		var me = this, items = [], data = App.stores.Picture.data.items;
695 
696 		// Get Images
697 		Ext.each(data, function(item, idx) {
698 			var htmlContent = '<div class="tapImage"><img width="'+lastpinch+'%" src="'+item.get('big_picture')+'"/></div>';
699 			if(!Ext.isEmpty(item.get('desc'))) {
700 				htmlContent += '<div class="description">'+item.get('desc')+'</div>';
701 			}
702 			items[idx] = new Ext.Panel({
703 				html: htmlContent,
704 				cls: 'slide_image',
705 				scroll: false,
706 				listeners: {
707 					scope: this,
708 					el: {
709 						delegate: '.tapImage',
710 						pinch: me.onPinch,
711 						doubletap: me.onDblTap
712 					}
713 				}
714 			});
715 		});
716 
717 		Ext.apply(me, {
718 			items: [items]
719 		});
720 		me.doLayout();
721 
722 		App.views.Shop.pictures.superclass.initComponent.call(this);
723 	},

725 	/**
726 	 * Resize the current picture through a pinch gesture
727 	 *
728 	 * @param obj
729 	 */
730 	onPinch: function(obj) {
731 		var element = this.query('img');
732 		Ext.each(element, function(el) {
733 
734 			// Calculate zoom value based on deltaScale
735 			lastpinch = lastpinch + (obj.deltaScale * 10);
736 			if(lastpinch >= 75) {
737 				if(lastpinch <= 120) {
738 					el.setAttribute('width', lastpinch + '%');
739 				} else {
740 					lastpinch = 120;
741 				}
742 			} else {
743 				lastpinch = 75;
744 			}
745 		});
746 	},
747 
748 	/**
749 	 * Resize the current picture to a specific value
750 	 */
751 	onDblTap: function() {
752 		if(lastpinch < 75) {
753 			return;
754 		}
755 		var element = this.query('img');
756 		Ext.each(element, function(el) {
757 			lastpinch = 75;
758 			el.setAttribute('width', lastpinch + '%');
759 		});
760 	}
761 });
