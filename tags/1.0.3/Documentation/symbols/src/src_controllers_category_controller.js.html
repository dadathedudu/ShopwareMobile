1 /**
  2  * @file category_controller.js
  3  * @link http://www.shopware.de
  4  * @author S.Pohl <stp@shopware.de>
  5  * @date 11-05-11
  6  */
  7 Ext.regController('category', {
  8 	view: Ext.getCmp('shop'),
  9 	store:    App.stores.Categories,
 10 	lastRecord: null,
 11 	
 12     show: function(options) {
 13 	    var record;
 14 
 15 	    if(!this.view) {
 16 		    this.view = Ext.getCmp('shop');
 17 	    }
 18 
 19 	    if(!Ext.isDefined(options.index) && !this.lastRecord) {
 20 		    this.view.setActiveItem(Ext.getCmp('home'), {
 21 				type: 'slide',
 22 				reverse: true,
 23 				scope: this
 24 			});
 25 		    this.view.toolBar.hide();
 26 		    this.view.doComponentLayout();
 27 		    return;
 28 	    }
 29 
 30 	    if(Ext.isDefined(options.store)) {
 31 		    record = options.store.getAt(options.index);
 32 		} else {
 33 		    record = this.store.getAt(options.index);
 34 	    }
 35 
 36 		if(!Ext.isDefined(options.index) && this.lastRecord) {
 37 			record = this.lastRecord;
 38 		}
 39 
 40 	    /* If no index and no last record, go back to home view */
 41 	    if(Ext.isEmpty(record)) {
 42 		    Ext.dispatch({
 43 			    controller: 'main',
 44 			    action: 'show',
 45 			    type: 'slide',
 46 			    direction: 'right',
 47 			    historyUrl: 'home'
 48 		    });
 49 	    } else {
 50 		    this.lastRecord = record;
 51 	    }
 52 
 53 	    if(Ext.isEmpty(record.data.sub)) {
 54 		    Ext.dispatch({
 55 			    controller:   'category',
 56 			    action:       'showArticleListing',
 57 			    categoryID:   record.data.id,
 58 			    categoryName: (!Ext.isEmpty(record.data.name)) ? record.data.name : record.data.text,
 59 			    type: (Ext.isDefined(options.type)) ? options.type : 'slide',
 60 			    direction: (Ext.isDefined(options.direction)) ? options.direction : 'left',
 61 			    panel: options.panel
 62 		    });
 63 	    } else {
 64 		    Ext.dispatch({
 65 			    controller:   'category',
 66 			    action:       'showSubCategories',
 67 			    categoryID:   Ext.isDefined(options.index) ? record.data.id : record.data.parentId,
 68 			    categoryName: record.data.name,
 69 			    type: (Ext.isDefined(options.type)) ? options.type : 'slide',
 70 			    direction: (Ext.isDefined(options.direction)) ? options.direction : 'left'
 71 		    });
 72 	    }
 73     },
 74 
 75 	showSubCategories: function(options) {
 76 		var store, type, direction, list = Ext.getCmp('subListing');
 77 	
 78 		if(!options.store) {
 79 			console.warn('options.store not defined, use default store');
 80 			store = App.stores.CategoriesTree;
 81 		} else {
 82 			store = options.store;
 83 		}
 84 		
 85 		if(!Ext.isDefined(options.categoryID)) {
 86 			throw new Error("No categoryID set in dispatch options");
 87 		}
 88 
 89 		if(!list) {
 90 			list = new App.views.Shop.subListing();
 91 			this.view.add(list);
 92 		}
 93 
 94 		if(Ext.isDefined(options.categoryName)) {
 95 			//this.view.toolBar.setTitle(options.categoryName);
 96 			list.title = options.categoryName;
 97 			list.toolbar.setTitle(options.categoryName);
 98 		}
 99 
100 		type = (Ext.isDefined(options.type)) ? options.type : 'slide';
101 		direction = (Ext.isDefined(options.direction)) ? options.direction : 'left';
102 
103 		this.view.setActiveItem(list, {type: type, direction: direction});
104 
105 		/* TODO - Needs a better workaround */
106 		list.setLoading(true);
107 		store.proxy.extraParams = {};
108 		window.setTimeout(function () {
109 			store.load({
110 				params: { categoryID: options.categoryID },
111 				callback: function() { list.setLoading(false) }
112 			});
113 		}, 150);
114 	},
115 	
116 	showArticleListing: function(options) {
117 		var store, list = Ext.getCmp('artListing'), type, direction;
118 		if(!options.store) {
119 			console.warn('options.store not defined, use default store');
120 			store = App.stores.Listing;
121 		} else {
122 			store = options.store;
123 		}
124 		
125 		if(!Ext.isDefined(options.categoryID)) {
126 			throw new Error("No categoryID set in dispatch options");
127 		}
128 
129 		if(!list) {
130 			list = new App.views.Shop.artListing;
131 			this.view.add(list);
132 		}
133 
134 		if(Ext.isDefined(options.categoryName)) {
135 			this.view.toolBar.setTitle(App.Helpers.truncate(options.categoryName, 10));
136 			this.view.title = options.categoryName;
137 			this.view.backBtn.show();
138 			this.view.toolBar.doLayout();
139 		}
140 
141 		type = (Ext.isDefined(options.type)) ? options.type : 'slide';
142 		direction = (Ext.isDefined(options.direction)) ? options.direction : 'left';
143 		
144 		this.view.setActiveItem(list, {type: type, direction: direction});
145 		store.proxy.extraParams = {};
146 		store.proxy.extraParams = { categoryID: options.categoryID };
147 		store.load({
148 			params: { categoryID: options.categoryID},
149 			callback: function() {
150 				store.fireEvent('checkBanner');
151 			}
152 		});
153 
154 		this.view.backBtn.show();
155 	    this.view.toolBar.show();
156 	    this.view.doComponentLayout();
157 	}
158 });
