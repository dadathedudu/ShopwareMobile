 95 	/**
 96 	 * Opens the detail view
 97 	 *
 98 	 * @param pnl
 99 	 * @param idx
100 	 */
101 	onItemTap: function(pnl, idx) {
102 		var item = App.stores.Info.getAt(idx), view = Ext.getCmp('infoDetail');
103 
104 		if(!view) {
105 			view = new App.views.Info.Detail(item);
106 			Ext.getCmp('info').add(view);
107 		}
108 
109 		Ext.getCmp('info').getToolbar().setTitle(App.Helpers.truncate(item.data.name, 12, '...'));
110 		Ext.getCmp('info').setActiveItem(view, {type: 'slide'});
111 	}

114 /**
115  * Detail Panel for list items
116  *
117  * @access public
118  * @class
119  * @extends Ext.form.FormPanel
120  */
121 App.views.Info.Detail = Ext.extend(Ext.form.FormPanel,
122 /** @lends App.views.Info.Detail# */
123 {
124 	id: 'infoDetail',
125 	height: '100%',
126 	scroll: 'vertical',
127 	url: '/sViewport,Forms',
128 	listeners: {
129 		scope: this,
130 		deactivate: function(me) {
131 			me.destroy();
132 		}
133 	},
134 
135 	/**
136 	 * Will be called when a new instance would create
137 	 *
138 	 * @param item
139 	 */
140 	constructor: function(item) {
141 
142 		if(Ext.isObject(item.data.form)) {
143 
144 			this.hidden = new Ext.form.Hidden({
145 				name: 'id',
146 				value: item.data.sFid
147 			});
148 
149 			this.submitHidden = new Ext.form.Hidden({
150 				name: 'Submit',
151 				value: 'Absenden'
152 			});
153 
154 			this.fieldset = new Ext.form.FieldSet({
155 				id: 'customFieldset',
156 				title: item.data.name,
157 				instructions: item.data.content,
158 				items: []
159 			});
160 			this.fieldset.removeAll();
161 
162 			for(var idx in item.data.form) {
163 				this.createFormElement(item.data.form[idx]);
164 			}
165 
166 			this.submitBtn = new Ext.Button({
167 				text: 'Absenden',
168 				ui: 'confirm',
169 				handler: function() {
170 					Ext.getCmp('infoDetail').submit();
171 				}
172 			})
173 
174 			Ext.apply(this, {
175 				items: [this.hidden, this.submitHidden, this.fieldset, this.submitBtn]
176 			});
177 			this.doLayout();
178 		} else {
179 			this.update('<div class="inner">' + item.data.content + '</div>');
180 		}
181 		App.views.Info.Detail.superclass.constructor.call(this);
182 	},
183 
184 	/**
185 	 * Dynamic creation of forms
186 	 * 
187 	 * @param cfg
188 	 */
189 	createFormElement: function(cfg) {
190 
191 		//Set references
192 		var defaultValue = '';
193 		if(cfg.ticket_task != null) {
194 			if(cfg.ticket_task == 'email') {
195 				defaultValue = this.userEmail;
196 			} else if(cfg.ticket_task == 'name') {
197 				defaultValue = this.userName;
198 			}
199 		}
200 		switch(cfg.typ) {
201 			case 'checkbox':
202 				var checkField = new Ext.form.Checkbox({
203 					label: cfg.label,
204 					name: cfg.name,
205 					checked: (parseInt(cfg.value)) ? true : false,
206 					required: (cfg.required) ? true : false
207 				});
208 				this.fieldset.add(checkField);
209 				break;
210 			case 'textarea':
211 				var areaField = new Ext.form.TextArea({
212 					label: cfg.label,
213 					name: cfg.name,
214 					allowBlank: (!cfg.required) ? true : false,
215 					required: (cfg.required) ? true : false,
216 					placeHolder: (cfg.note) ? cfg.note : false
217 				});
218 				this.fieldset.add(areaField);
219 				break;
220 
221 			case 'email':
222 				var mailField = new Ext.form.Email({
223 					label: 'Mail',
224 					name: cfg.name,
225 					value: defaultValue,
226 					allowBlank: (!cfg.required) ? true : false,
227 					required: (cfg.required) ? true : false,
228 					placeHolder: (cfg.note) ? cfg.note : false
229 				});
230 				this.fieldset.add(mailField);
231 				break;
232 			
233 			case 'select':
234 				var selectValues = cfg.value.split(';');
235 				var data = new Array();
236 				if(selectValues != null) {
237 					for(var i=0; i<selectValues.length; i++) {
238 						var tmpArr = new Array();
239 						tmpArr.push(selectValues[i]);
240 						data.push(tmpArr);
241 					}
242 				}
243 
244 				var selectField = new Ext.form.Select({
245 					label: cfg.label,
246 					name: cfg.name,
247 					hiddenName: cfg.name,
248 					mode: 'local',
249 					displayField: 'value',
250 					valueField: 'value',
251 					store: new Ext.data.ArrayStore({ autoDestroy: true, fields: ['value'], data: data }),
252 					triggerAction: 'all',
253 					required: (cfg.required) ? true : false
254 				});
255 				this.fieldset.add(selectField);
256 				break;
257 			
258 			default:
259 				var textField = new Ext.form.Text({
260 					label: cfg.label,
261 					name: cfg.name,
262 					value: defaultValue,
263 					allowBlank: (!cfg.required) ? true : false,
264 					required: (cfg.required) ? true : false,
265 					placeHolder: (cfg.note) ? cfg.note : false
266 				});
267 			this.fieldset.add(textField);
268 		}
269 	}
270 });
